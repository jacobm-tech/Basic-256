# Contributed by Joel Kahn
fastgraphics
cls
clg
c=7
xs=750
ys=520
graphsize xs,ys
for y=ys to 0 step -1
   for x=0 to xs
      c=c*1.00005
      color rgb(c\256\256%256, c\256%256, c%256)
      plot x,y
   next x
   refresh
next y
print c
